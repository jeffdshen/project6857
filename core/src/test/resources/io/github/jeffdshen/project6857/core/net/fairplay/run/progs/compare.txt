/*
 * Compare two pieces and see who wins
 *
 */
program Compare {
    type int = Int<4>; // 4-bit integer
    type Rank = Int<3>; // only need 3 bits - note that it starts at 0, not 1.
    type PieceType = enum { ROCK, PAPER, SCISSORS }; // flag should never be reached
    type CompareResult = enum { WIN, LOSS, TIE };

	type AliceInput = struct { PieceType piece, Rank rank };
	type BobInput = struct { PieceType piece, Rank rank };
	type AliceOutput = CompareResult;
	type BobOutput = CompareResult;
	type Output = struct {AliceOutput alice,
            BobOutput bob};
	type Input = struct {AliceInput alice,
            BobInput bob};

	function Output output(Input input) {
	    // if alice has a bomb
	    if (input.alice.rank == 0) {
            if (input.alice.piece == ROCK) {
                if (input.bob.piece == PAPER) {
                    output.alice = LOSS;
                    output.bob = WIN;
                } else {
                    output.alice = WIN;
                    output.bob = LOSS;
                }
            } else if (input.alice.piece == PAPER) {
                if (input.bob.piece == SCISSORS) {
                    output.alice = LOSS;
                    output.bob = WIN;
                } else {
                     output.alice = WIN;
                     output.bob = LOSS;
                }
            } else {
                if (input.bob.piece == ROCK) {
                    output.alice = LOSS;
                    output.bob = WIN;
                } else {
                    output.alice = WIN;
                    output.bob = LOSS;
                }
            }
	    } else if (input.bob.rank == 0) {
            if (input.bob.piece == ROCK) {
                if (input.alice.piece == PAPER) {
                    output.bob = LOSS;
                    output.alice = WIN;
                } else {
                    output.bob = WIN;
                    output.alice = LOSS;
                }
            } else if (input.bob.piece == PAPER) {
                if (input.alice.piece == SCISSORS) {
                    output.bob = LOSS;
                    output.alice = WIN;
                } else {
                     output.bob = WIN;
                     output.alice = LOSS;
                }
            } else {
                if (input.alice.piece == ROCK) {
                    output.bob = LOSS;
                    output.alice = WIN;
                } else {
                    output.bob = WIN;
                    output.alice = LOSS;
                }
            }
	    } else if (input.alice.piece == ROCK) {
            if (input.bob.piece == ROCK) {
                if (input.alice.rank > input.bob.rank) {
                    output.alice = WIN;
                    output.bob = LOSS;
                } else if (input.alice.rank < input.bob.rank) {
                    output.alice = LOSS;
                    output.bob = WIN;
                } else {
                    output.alice = TIE;
                    output.bob = TIE;
                }
            } else if (input.bob.piece == PAPER) {
                if (input.alice.rank > input.bob.rank + 2) {
                    output.alice = WIN;
                    output.bob = LOSS;
                } else if (input.alice.rank < input.bob.rank + 2) {
                    output.alice = LOSS;
                    output.bob = WIN;
                } else {
                    output.alice = TIE;
                    output.bob = TIE;
                }
            } else {
                if (input.alice.rank + 2 > input.bob.rank) {
                    output.alice = WIN;
                    output.bob = LOSS;
                } else if (input.alice.rank + 2 < input.bob.rank) {
                    output.alice = LOSS;
                    output.bob = WIN;
                } else {
                    output.alice = TIE;
                    output.bob = TIE;
                }
            }
        } else if (input.alice.piece == PAPER) {
            if (input.bob.piece == ROCK) {
                if (input.alice.rank + 2 > input.bob.rank) {
                    output.alice = WIN;
                    output.bob = LOSS;
                } else if (input.alice.rank + 2 < input.bob.rank) {
                    output.alice = LOSS;
                    output.bob = WIN;
                } else {
                    output.alice = TIE;
                    output.bob = TIE;
                }
            } else if (input.bob.piece == PAPER) {
                if (input.alice.rank > input.bob.rank) {
                    output.alice = WIN;
                    output.bob = LOSS;
                } else if (input.alice.rank < input.bob.rank) {
                    output.alice = LOSS;
                    output.bob = WIN;
                } else {
                    output.alice = TIE;
                    output.bob = TIE;
                }
            } else {
                if (input.alice.rank > input.bob.rank + 2) {
                    output.alice = WIN;
                    output.bob = LOSS;
                } else if (input.alice.rank < input.bob.rank + 2) {
                    output.alice = LOSS;
                    output.bob = WIN;
                } else {
                    output.alice = TIE;
                    output.bob = TIE;
                }
            }
        } else {
            if (input.bob.piece == ROCK) {
                if (input.alice.rank > input.bob.rank + 2) {
                    output.alice = WIN;
                    output.bob = LOSS;
                } else if (input.alice.rank < input.bob.rank + 2) {
                    output.alice = LOSS;
                    output.bob = WIN;
                } else {
                    output.alice = TIE;
                    output.bob = TIE;
                }
            } else if (input.bob.piece == PAPER) {
                if (input.alice.rank + 2 > input.bob.rank) {
                    output.alice = WIN;
                    output.bob = LOSS;
                } else if (input.alice.rank + 2 < input.bob.rank) {
                    output.alice = LOSS;
                    output.bob = WIN;
                } else {
                    output.alice = TIE;
                    output.bob = TIE;
                }
            } else {
                if (input.alice.rank > input.bob.rank) {
                    output.alice = WIN;
                    output.bob = LOSS;
                } else if (input.alice.rank < input.bob.rank) {
                    output.alice = LOSS;
                    output.bob = WIN;
                } else {
                    output.alice = TIE;
                    output.bob = TIE;
                }
            }
        }
	}
}
